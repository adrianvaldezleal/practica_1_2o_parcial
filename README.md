#almacenar una lista de precios
list1 = [43, 57, 92, 20, 37, 5, 9]
list2 = [43, 57, 92, 20, 37, 5, 9] 

#identificar el precio mínimo
precio1 = min(list1)

precio = max(list2)

#colocar los números para enumerarlos de menor a mayor

thislist.sort()

print("Lista ordenada:", thislist)
